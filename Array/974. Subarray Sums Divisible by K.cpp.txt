class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        int n = nums.size();
        unordered_map<int,int> mp;
        mp[0]++;
        int cnt = 0;
        int currSum = 0;
        for(int i = 0; i<n; i++) {
            currSum = (currSum + nums[i]%k + k)%k;
            cout<<currSum<<" "<<mp[currSum]<<endl;
            cnt += mp[currSum];
            mp[currSum]++;
        }           
        return cnt;
    }
};
